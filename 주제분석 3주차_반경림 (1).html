    layers.Dense(256, activation = 'relu', input_shape = [len(train_tune_cbe.keys())]),

    layers.Dense(32, activation = 'relu'),
    layers.Dense(1)
  ])

  optimizer = tf.keras.optimizers.RMSprop(0.0001)

  model.compile(loss = 'mse',
                optimizer = optimizer,
                metrics = ['mse'])
  return model

# Sequential 모델은 레이어들이 일렬로 쭉 나열된 형태이다.
# 첫번째 레이어는 인풋레이어로 입력 노드 값이 8, 출력 노드 값이 256이다.
# 두번째 레이어의 입력값은 앞의 레이어의 출력값이므로 256, 출력 값은 64다.
# 파라미터는 각 입력노드와 출력노드에 대해 연결된 간선의 수를 의미한다.
# 첫번째 레이어에는 입력값(8)+바이어스노드(1)=9개의 인풋 노드가 있고, 파라미터는 9*256=2304개인 것이다.
# 마찬가지로 두번째 레이어에는 (256+1)*64, 세번째는 (64+1)*32, 네번째는 (32+1)*1개의 파라미터가 존재한다.
